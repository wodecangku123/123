    <script>
        (function() {
            // ==================== 配置 ====================
            const ADMIN_PASSWORD = 'admin123'; // 修改管理员密码
            const DEFAULT_PRIZES = [
                { name: '一等奖', weight: 2, image: '' },
                { name: '二等奖', weight: 5, image: '' },
                { name: '三等奖', weight: 8, image: '' },
                { name: '四等奖', weight: 15, image: '' },
                { name: '参与奖', weight: 25, image: '' },
                { name: '谢谢参与', weight: 35, image: '' }
            ];
            const COLORS = ['#FF6B6B','#4ECDC4','#45B7D1','#96CEB4','#FFEAA7','#DDA0DD',
                           '#98D8C8','#F7DC6F','#FF8C42','#A29BFE','#FD79A8','#00B894'];

            // ==================== 全局状态 ====================
            let prizes = [...DEFAULT_PRIZES];
            let loadedImages = {};
            let currentRotation = 0;
            let isSpinning = false;
            let isAdminLoggedIn = false;
            let records = [];
            let limitConfig = {
                enabled: false,
                perUser: 3,
                total: 100
            };
            let globalDrawCount = 0;

            // ==================== 工具函数 ====================
            function getUserId() {
                let id = localStorage.getItem('wheel_user_id');
                if (!id) {
                    id = 'u_' + Date.now() + Math.random().toString(36).substr(2,6);
                    localStorage.setItem('wheel_user_id', id);
                }
                return id;
            }

            function getUserDrawCount() {
                const key = 'user_draws_' + getUserId();
                return parseInt(localStorage.getItem(key) || '0');
            }

            function setUserDrawCount(count) {
                const key = 'user_draws_' + getUserId();
                localStorage.setItem(key, count.toString());
            }

            function loadAllData() {
                // 加载奖项
                const savedPrizes = localStorage.getItem('wheel_prizes');
                if (savedPrizes) {
                    try { prizes = JSON.parse(savedPrizes); } catch(e) {}
                }
                // 加载记录
                const savedRecords = localStorage.getItem('wheel_records');
                if (savedRecords) {
                    try { records = JSON.parse(savedRecords); } catch(e) {}
                }
                // 加载限制配置
                const savedConfig = localStorage.getItem('wheel_limit_config');
                if (savedConfig) {
                    try { limitConfig = JSON.parse(savedConfig); } catch(e) {}
                }
                // 加载全局计数器
                globalDrawCount = parseInt(localStorage.getItem('wheel_global_count') || '0');
            }

            function saveAll() {
                localStorage.setItem('wheel_prizes', JSON.stringify(prizes));
                localStorage.setItem('wheel_records', JSON.stringify(records));
                localStorage.setItem('wheel_limit_config', JSON.stringify(limitConfig));
                localStorage.setItem('wheel_global_count', globalDrawCount.toString());
            }

            // ==================== 转盘绘制 ====================
            async function cacheImages() {
                const promises = prizes.map(p => {
                    if (!p.image) return Promise.resolve(null);
                    if (loadedImages[p.image]) return Promise.resolve(loadedImages[p.image]);
                    return new Promise(resolve => {
                        const img = new Image();
                        img.crossOrigin = 'anonymous';
                        img.onload = () => { loadedImages[p.image] = img; resolve(img); };
                        img.onerror = () => resolve(null);
                        img.src = p.image;
                    });
                });
                await Promise.all(promises);
            }

            function drawWheel() {
                if (!prizes.length) return;
                const canvas = document.getElementById('wheelCanvas');
                const ctx = canvas.getContext('2d');
                const size = Math.min(window.innerWidth * 0.85, 450);
                canvas.width = size * 2;
                canvas.height = size * 2;
                canvas.style.width = size + 'px';
                canvas.style.height = size + 'px';
                const w = canvas.width, h = canvas.height;
                const cx = w/2, cy = h/2, r = w/2 - 16;
                const sliceAngle = (2*Math.PI) / prizes.length;

                ctx.clearRect(0,0,w,h);
                const fontSize = Math.min(36, Math.max(20, 280/prizes.length + 12));
                const imgSize = Math.min(60, Math.max(32, 200/prizes.length + 16));

                prizes.forEach((prize, i) => {
                    const start = i*sliceAngle - Math.PI/2;
                    const end = start + sliceAngle;
                    ctx.beginPath();
                    ctx.moveTo(cx, cy);
                    ctx.arc(cx, cy, r, start, end);
                    ctx.fillStyle = COLORS[i % COLORS.length];
                    ctx.fill();
                    ctx.strokeStyle = '#fff';
                    ctx.lineWidth = 4;
                    ctx.stroke();

                    ctx.save();
                    ctx.translate(cx, cy);
                    ctx.rotate(start + sliceAngle/2);
                    const img = loadedImages[prize.image];
                    if (img && prizes.length <= 25) {
                        const imgX = r - imgSize - 40;
                        ctx.drawImage(img, imgX, -imgSize/2, imgSize, imgSize);
                    }
                    ctx.fillStyle = '#fff';
                    ctx.font = `bold ${fontSize}px "Microsoft YaHei"`;
                    ctx.textAlign = 'right';
                    ctx.shadowColor = 'rgba(0,0,0,0.5)';
                    ctx.shadowBlur = 4;
                    ctx.fillText(prize.name, r - 20, fontSize/3);
                    ctx.restore();
                });
            }

            // ==================== 抽奖逻辑 ====================
            function updateRemainInfo() {
                const infoEl = document.getElementById('remainInfo');
                if (!limitConfig.enabled) {
                    infoEl.textContent = '';
                    return;
                }
                const userRemain = limitConfig.perUser > 0 ? limitConfig.perUser - getUserDrawCount() : Infinity;
                const totalRemain = limitConfig.total > 0 ? limitConfig.total - globalDrawCount : Infinity;
                const remain = Math.min(userRemain, totalRemain);
                if (remain <= 0) {
                    infoEl.textContent = '🚫 抽奖次数已用完';
                    document.getElementById('spinBtn').disabled = true;
                } else {
                    infoEl.textContent = `剩余抽奖次数：${remain}`;
                    document.getElementById('spinBtn').disabled = false;
                }
            }

            function canDraw() {
                if (!limitConfig.enabled) return true;
                const userCount = getUserDrawCount();
                if (limitConfig.perUser > 0 && userCount >= limitConfig.perUser) return false;
                if (limitConfig.total > 0 && globalDrawCount >= limitConfig.total) return false;
                return true;
            }

            function spin() {
                if (isSpinning || prizes.length === 0) return;
                if (!canDraw()) {
                    alert('抽奖次数已用完！');
                    return;
                }

                isSpinning = true;
                document.getElementById('spinBtn').disabled = true;
                document.getElementById('resultPanel').innerHTML = '🎰 转动中...';

                // 更新次数
                if (limitConfig.enabled) {
                    setUserDrawCount(getUserDrawCount() + 1);
                    globalDrawCount++;
                    saveAll();
                    updateRemainInfo();
                }

                // 权重随机
                const totalWeight = prizes.reduce((s,p) => s + p.weight, 0);
                let rand = Math.random() * totalWeight;
                let winIndex = 0;
                for (let i = 0; i < prizes.length; i++) {
                    rand -= prizes[i].weight;
                    if (rand <= 0) { winIndex = i; break; }
                }

                const sliceAngle = 360 / prizes.length;
                const prizeCenter = winIndex * sliceAngle + sliceAngle/2;
                const targetFinal = (360 - prizeCenter) % 360;
                const currentDeg = ((currentRotation % 360) + 360) % 360;
                let delta = targetFinal - currentDeg;
                if (delta < 0) delta += 360;
                const extra = (6 + Math.random()*3) * 360;
                currentRotation += extra + delta;
                document.getElementById('wheelRotateLayer').style.transform = `rotate(${currentRotation}deg)`;

                setTimeout(() => {
                    const prize = prizes[winIndex];
                    // 记录
                    records.push({
                        prize: prize.name,
                        image: prize.image,
                        time: new Date().toISOString(),
                        userId: getUserId()
                    });
                    if (records.length > 1000) records = records.slice(-1000);
                    saveAll();

                    document.getElementById('resultPanel').innerHTML = 
                        (prize.image ? `<img src="${prize.image}">` : '') + `🎉 恭喜获得：${prize.name}`;
                    isSpinning = false;
                    updateRemainInfo();
                }, 5200);
            }

            // ==================== 管理面板 ====================
            function renderAdminPrizes() {
                const container = document.getElementById('prizeListContainer');
                container.innerHTML = prizes.map((p,i) => `
                    <div class="prize-item">
                        <input value="${p.name}" placeholder="名称" class="pname">
                        <input type="number" value="${p.weight}" min="0" step="0.1" style="width:70px" class="pweight">
                        <input value="${p.image || ''}" placeholder="图片URL" class="pimage">
                        <button class="btn btn-danger" onclick="this.parentElement.remove()">删</button>
                    </div>`).join('');
                updateTotalWeight();
            }

            function updateTotalWeight() {
                let total = 0;
                document.querySelectorAll('.pweight').forEach(el => total += parseFloat(el.value)||0);
                document.getElementById('totalWeightDisplay').textContent = total.toFixed(1);
            }

            function savePrizesFromUI() {
                const items = document.querySelectorAll('.prize-item');
                prizes = [];
                items.forEach(el => {
                    const name = el.querySelector('.pname').value.trim();
                    const weight = parseFloat(el.querySelector('.pweight').value)||0;
                    const image = el.querySelector('.pimage').value.trim();
                    if (name) prizes.push({name, weight, image});
                });
                if (prizes.length < 2) { alert('至少需要2个奖项'); return; }
                saveAll();
                cacheImages().then(drawWheel);
                alert('奖项已保存');
            }

            function renderRecords() {
                const list = document.getElementById('recordsList');
                if (records.length === 0) {
                    list.innerHTML = '暂无记录';
                    return;
                }
                const recent = [...records].reverse().slice(0, 100);
                list.innerHTML = recent.map((r,i) => {
                    const time = new Date(r.time).toLocaleString('zh-CN');
                    return `<div style="padding:8px;border-bottom:1px solid #eee;">
                        <strong>${r.prize}</strong> - ${time} (用户:${r.userId.slice(-6)})
                    </div>`;
                }).join('');
            }

            function exportCSV() {
                if (records.length === 0) return alert('无记录');
                let csv = '奖项,时间,用户ID\n';
                records.forEach(r => {
                    csv += `"${r.prize}","${new Date(r.time).toLocaleString('zh-CN')}","${r.userId}"\n`;
                });
                const blob = new Blob(['\uFEFF'+csv], {type:'text/csv'});
                const a = document.createElement('a');
                a.href = URL.createObjectURL(blob);
                a.download = `抽奖记录_${new Date().toLocaleDateString()}.csv`;
                a.click();
            }

            function clearRecords() {
                if (confirm('确定清空所有记录？')) {
                    records = [];
                    saveAll();
                    renderRecords();
                }
            }

            function loadLimitUI() {
                document.getElementById('limitEnabled').checked = limitConfig.enabled;
                document.getElementById('perUserLimit').value = limitConfig.perUser;
                document.getElementById('totalLimit').value = limitConfig.total;
                document.getElementById('currentTotalDraws').textContent = globalDrawCount;
            }

            function saveLimitFromUI() {
                limitConfig.enabled = document.getElementById('limitEnabled').checked;
                limitConfig.perUser = parseInt(document.getElementById('perUserLimit').value) || 0;
                limitConfig.total = parseInt(document.getElementById('totalLimit').value) || 0;
                saveAll();
                updateRemainInfo();
                alert('限制设置已保存');
            }

            function resetAllUserCounts() {
                if (confirm('确定重置所有用户的抽奖次数吗？（不影响总计数器）')) {
                    // 清除所有用户的本地计数（只能清除当前浏览器的，其他用户下次抽奖时会重新计数）
                    localStorage.removeItem('user_draws_' + getUserId());
                    alert('当前用户的次数已重置（其他用户需自行清除浏览器数据或联系管理员）');
                }
            }

            function resetGlobalCount() {
                if (confirm('确定将总抽奖次数计数器归零吗？')) {
                    globalDrawCount = 0;
                    saveAll();
                    document.getElementById('currentTotalDraws').textContent = '0';
                    updateRemainInfo();
                    alert('总计数器已重置');
                }
            }

            // ==================== 密码验证 ====================
            function showPasswordModal() {
                document.getElementById('passwordModal').classList.add('active');
                document.getElementById('passwordInput').value = '';
                document.getElementById('passwordError').style.display = 'none';
                document.getElementById('passwordInput').focus();
            }

            function closePasswordModal() {
                document.getElementById('passwordModal').classList.remove('active');
            }

            function verifyPassword() {
                if (document.getElementById('passwordInput').value === ADMIN_PASSWORD) {
                    isAdminLoggedIn = true;
                    closePasswordModal();
                    openAdminPanel();
                } else {
                    document.getElementById('passwordError').style.display = 'block';
                }
            }

            function openAdminPanel() {
                if (!isAdminLoggedIn) return showPasswordModal();
                renderAdminPrizes();
                renderRecords();
                loadLimitUI();
                document.getElementById('adminPanel').classList.add('active');
            }

            function closeAdminPanel() {
                document.getElementById('adminPanel').classList.remove('active');
            }

            // ==================== 事件绑定 ====================
            document.getElementById('spinBtn').addEventListener('click', spin);
            document.getElementById('title').addEventListener('click', function() {
                if (isAdminLoggedIn) openAdminPanel();
                else showPasswordModal();
            });
            document.getElementById('confirmPasswordBtn').addEventListener('click', verifyPassword);
            document.getElementById('cancelPasswordBtn').addEventListener('click', closePasswordModal);
            document.getElementById('closeAdminBtn').addEventListener('click', closeAdminPanel);

            // 标签切换
            document.querySelectorAll('.tab-btn').forEach(btn => {
                btn.addEventListener('click', function() {
                    document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
                    this.classList.add('active');
                    const tab = this.dataset.tab;
                    document.querySelectorAll('.tab-content').forEach(c => c.classList.remove('active'));
                    document.getElementById(tab + 'Tab').classList.add('active');
                    if (tab === 'records') renderRecords();
                });
            });

            document.getElementById('addPrizeBtn').addEventListener('click', () => {
                const container = document.getElementById('prizeListContainer');
                const div = document.createElement('div');
                div.className = 'prize-item';
                div.innerHTML = `<input value="新奖项" class="pname"><input type="number" value="1" class="pweight" style="width:70px"><input placeholder="图片URL" class="pimage"><button class="btn btn-danger" onclick="this.parentElement.remove()">删</button>`;
                container.appendChild(div);
                updateTotalWeight();
            });

            document.getElementById('prizeListContainer').addEventListener('input', updateTotalWeight);
            document.getElementById('savePrizesBtn').addEventListener('click', savePrizesFromUI);
            document.getElementById('resetDefaultBtn').addEventListener('click', () => {
                if (confirm('恢复默认奖项？')) {
                    prizes = [...DEFAULT_PRIZES];
                    saveAll();
                    cacheImages().then(drawWheel);
                    renderAdminPrizes();
                }
            });

            document.getElementById('exportRecordsBtn').addEventListener('click', exportCSV);
            document.getElementById('clearRecordsBtn').addEventListener('click', clearRecords);
            document.getElementById('saveLimitBtn').addEventListener('click', saveLimitFromUI);
            document.getElementById('resetAllCountsBtn').addEventListener('click', resetAllUserCounts);
            document.getElementById('resetGlobalCountBtn').addEventListener('click', resetGlobalCount);

            // 初始化
            loadAllData();
            cacheImages().then(() => {
                drawWheel();
                updateRemainInfo();
            });
        })();
